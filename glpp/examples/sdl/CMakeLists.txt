###############################################################################
#   Find and store SDL paths.
###############################################################################
find_package( SDL )


###############################################################################
#   Set include directories.
###############################################################################
set( INCLUDES
        ${SDL_INCLUDE_DIR}
        ${GLPP_ROOT}/includes )

include_directories( ${INCLUDES} )


###############################################################################
#   Set sources.
###############################################################################
set( SOURCES
        sdl_example.cpp )


###############################################################################
#   Build shim library.
###############################################################################
add_executable( glpp_sdl_example ${SOURCES} ${HEADERS} )
target_link_libraries( glpp_sdl_example glpp ${SDL_LIBRARY} ${ALL_OPENGL_LIBRARIES} )
